#include <REGX51.H>

void delay(unsigned int time) {
    unsigned int i, j;
    for(i = 0; i < time; i++)
        for(j = 0; j < 1275; j++);
}

void main() {
    while(1) {
        P1 = 0xAA;   // 10101010 – even LEDs ON
        delay(500);
        P1 = 0x55;   // 01010101 – odd LEDs ON
        delay(500);
    }
}
